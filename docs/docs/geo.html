 32def dms2rad(deg, min=0, sec=0):
 33    """Convert degrees, minutes, seconds to radians.
 34
 35    Parameters
 36    ----------
 37    deg: array_like
 38        Angle in degrees.
 39    min: array_like
 40        (optional) Angle component in minutes.
 41    sec: array_like
 42        (optional) Angle component in seconds.
 43    Returns
 44    -------
 45    numpy.ndarray
 46        Angle in radians.
 47    """
 48    deg = asarray(deg)
 49    # In a full circle there are 360 degrees(°).
 50    # Each degree is split up into 60 parts, each part being 1/60 of a degree. These parts are called minutes(').
 51    # Each minute is split up into 60 parts, each part being 1/60 of a minute. These parts are called seconds('').
 52    # 1°=60′，1′=60″ ，1°=3600″
 53    return deg2rad(deg + min / 60. + sec / 3600.)

 56def rad2dms(rad, dms=False):
 57    """Convert radians to degrees, minutes, seconds.
 58    Parameters
 59    ----------
 60    rad: array_like
 61        Angle in radians.
 62    dms: bool
 63        Use degrees, minutes, seconds format. If False, use decimal degrees.
 64    Returns
 65    -------
 66    numpy.ndarray
 67        Angle in degrees, minutes, seconds or decimal degrees.
 68    """
 69
 70    rad = asarray(rad)
 71    deg = rad2deg(rad)
 72    if dms:
 73        min = 60.0 * mod(deg, 1.0)
 74        sec = 60.0 * mod(min, 1.0)
 75        return stack((floor(deg), floor(min), sec.round(4)))
 76    else:
 77        return deg

 94def lat_long_to_xyz(phi, lam, rads=False, datum=osgb36):
 95    """Convert input latitude/longitude in a given datum into
 96    Cartesian (x, y, z) coordinates.
 97    Parameters
 98    ----------
 99    phi: array_like
100        Latitude in degrees (if radians=False) or radians (if radians=True).
101    lam: array_like
102        Longitude in degrees (if radians=False) or radians (if radians=True).
103    rads: bool (optional)
104        If True, input latitudes and longitudes are in radians.
105    datum: Datum (optional)
106        Datum to use for conversion.
107    """
108    if not rads:
109        phi = deg2rad(phi)
110        lam = deg2rad(lam)
111
112    nu = datum.a * datum.F_0 / sqrt(1 - datum.e2 * sin(phi) ** 2)
113
114    return array(((nu + datum.H) * cos(phi) * cos(lam),
115                  (nu + datum.H) * cos(phi) * sin(lam),
116                  ((1 - datum.e2) * nu + datum.H) * sin(phi)))
117
118def xyz_to_lat_long(x, y, z, rads=False, datum=osgb36):
119    p = sqrt(x ** 2 + y ** 2)
120
121    lam = arctan2(y, x)
122    phi = arctan2(z, p * (1 - datum.e2))
123
124    for _ in range(10):
125        nu = datum.a * datum.F_0 / sqrt(1 - datum.e2 * sin(phi) ** 2)
126        dnu = -datum.a * datum.F_0 * cos(phi) * sin(phi) / (1 - datum.e2 * sin(phi) ** 2) ** 1.5
127
128        f0 = (z + datum.e2 * nu * sin(phi)) / p - tan(phi)
129        f1 = datum.e2 * (nu ** cos(phi) + dnu * sin(phi)) / p - 1.0 / cos(phi) ** 2
130        phi -= f0 / f1
131
132    if not rads:
133        phi = rad2dms(phi)
134        lam = rad2dms(lam)
135
136    return phi, lam

139def get_easting_northing_from_gps_lat_long(phi, lam, rads=False):
140    """ Get OSGB36 easting/northing from GPS latitude and longitude pairs.
141
142    Parameters
143    ----------
144    phi: float/arraylike
145        GPS (i.e. WGS84 datum) latitude value(s)
146    lam: float/arrayling
147        GPS (i.e. WGS84 datum) longitude value(s).
148    rads: bool (optional)
149        If true, specifies input is is radians.
150    Returns
151    -------
152    numpy.ndarray
153        Easting values (in m)
154    numpy.ndarray
155        Northing values (in m)
156
157    Examples
158    --------
159    >>> get_easting_northing_from_gps_lat_long([55.5], [-1.54])
160    (array([429157.5449526]), array([623009.09706685]))
161
162    References
163    ----------
164    Based on the formulas in "A guide to coordinate systems in Great Britain".
165    See also https://webapps.bgs.ac.uk/data/webservices/convertForm.cfm
166    """
167    # from WGS84 to OSGB36
168
169    phi, lam = WGS84toOSGB36(phi, lam, rads=False)
170    phi = dms2rad(phi)
171    lam = dms2rad(lam)
172
173    a = osgb36.a
174    b = osgb36.b
175    n = osgb36.n
176    e2 = osgb36.e2
177    F_0 = osgb36.F_0
178    N_0 = osgb36.N_0
179    E_0 = osgb36.E_0
180    phi_0 = osgb36.phi_0
181    lam_0 = osgb36.lam_0
182
183    nu = a * F_0 * (1 - e2 * sin(phi) ** 2) ** -0.5
184    rho = a * F_0 * (1 - e2) * (1 - e2 * sin(phi) ** 2) ** -1.5
185    eta2 = nu / rho - 1
186
187    M = b * F_0 * ((1 + n + (5 / 4) * (n ** 2 + n ** 3)) * (phi - phi_0)
188                   - (3 * (n + n ** 2) + (21 / 8) * n ** 3) * sin(phi - phi_0) * cos(phi + phi_0)
189                   + ((15 / 8) * (n ** 2 + n ** 3) * sin(2 * (phi - phi_0)) * cos(2 * (phi + phi_0)))
190                   - (35 / 24) * n ** 3 * sin(3 * (phi - phi_0)) * cos(3 * (phi + phi_0)))
191
192    I = M + N_0
193    II = nu / 2 * sin(phi) * cos(phi)
194    III = nu / 24 * sin(phi) * cos(phi) ** 3 * (5 - tan(phi) ** 2 + 9 * eta2)
195    IIIA = nu / 720 * sin(phi) * cos(phi) ** 5 * (61 - 58 * tan(phi) ** 2 + tan(phi) ** 4)
196    IV = nu * cos(phi)
197    V = nu / 6 * cos(phi) ** 3 * (nu / rho - tan(phi) ** 2)
198    VI = nu / 120 * cos(phi) ** 5 * (5 - 18 * tan(phi) ** 2 + tan(phi) ** 4 + 14 * eta2 - 58 * tan(phi) ** 2 * eta2)
199
200    N = I + II * (lam - lam_0) ** 2 + III * (lam - lam_0) ** 4 + IIIA * (lam - lam_0) ** 6
201
202    E = E_0 + IV * (lam - lam_0) + V * (lam - lam_0) ** 3 + VI * (lam - lam_0) ** 5
203
204    return E,N

243    e2 = osgb36.e2
244    F_0 = osgb36.F_0
245    N_0 = osgb36.N_0
246    E_0 = osgb36.E_0

251    my_lam = []

254        E = east[i]
255        N = north[i]

258
259        M = b * F_0 * ((1 + n + (5 / 4) * (n ** 2 + n ** 3)) * (phi - phi_0)
260                       - (3 * (n + n ** 2) + (21 / 8) * n ** 3) * sin(phi - phi_0) * cos(phi + phi_0)
261                       + ((15 / 8) * (n ** 2 + n ** 3) * sin(2 * (phi - phi_0)) * cos(2 * (phi + phi_0)))
262                       - (35 / 24) * n ** 3 * sin(3 * (phi - phi_0)) * cos(3 * (phi + phi_0)))
263
264        while abs(N - N_0 - M) >= 0.00001:
265            phi = (N - N_0 - M) / (a * F_0) + phi
266
267            M = b * F_0 * ((1 + n + (5 / 4) * (n ** 2 + n ** 3)) * (phi - phi_0)
268                           - (3 * (n + n ** 2) + (21 / 8) * n ** 3) * sin(phi - phi_0) * cos(phi + phi_0)
269                           + ((15 / 8) * (n ** 2 + n ** 3) * sin(2 * (phi - phi_0)) * cos(2 * (phi + phi_0)))
270                           - (35 / 24) * n ** 3 * sin(3 * (phi - phi_0)) * cos(3 * (phi + phi_0)))
271
272        nu = a * F_0 * (1 - e2 * sin(phi) ** 2) ** -0.5
273        rho = a * F_0 * (1 - e2) * (1 - e2 * sin(phi) ** 2) ** -1.5
274        eta2 = nu / rho - 1
275
276        VII = tan(phi) / (2 * rho * nu)
277        VIII = tan(phi) / (24 * rho * nu ** 3) * (5 + 3 * tan(phi) ** 2 + eta2 - 9 * tan(phi) ** 2 * eta2)
278        IX = tan(phi) / (720 * rho * nu ** 5) * (61 + 90 * tan(phi) ** 2 + 45 * tan(phi) ** 4)
279        X = 1 / (nu * cos(phi))
280        XI = (nu / rho + 2 * tan(phi) ** 2) / (cos(phi) * 6 * nu ** 3)

285        lam = lam_0 + X * (E - E_0) - XI * (E - E_0) ** 3 + XII * (E - E_0) ** 5 - XIIA * (E - E_0) ** 7
286        phi, lam = OSGB36toWGS84(phi, lam, rads=True)
287        my_phi.append(rad2dms(phi)[0])
288        my_lam.append(rad2dms(lam)[0])
289
290    return np.array(my_phi), np.array(my_lam)

293class HelmertTransform(object):
294    """Callable class to perform a Helmert transform."""
295
296    def __init__(self, s, rx, ry, rz, T):
297        self.T = T.reshape((3, 1))
298
299        self.M = array([[1 + s, -rz, ry],
300                        [rz, 1 + s, -rx],
301                        [-ry, rx, 1 + s]])
302
303    def __call__(self, X):
304        X = X.reshape((3, -1))
305        return self.T + self.M @ X
306
307
308class HelmertInverseTransform(object):
309    """Callable class to perform the inverse of a Helmert transform."""
310
311    def __init__(self, s, rx, ry, rz, T):
312        self.T = T.reshape((3, 1))
313
314        self.M = inv(array([[1 + s, -rz, ry],
315                            [rz, 1 + s, -rx],
316                            [-ry, rx, 1 + s]]))
317
318    def __call__(self, X):
319        X = X.reshape((3, -1))
320        return self.M @ (X - self.T)

281        XI = (nu / rho + 2 * tan(phi) ** 2) / (cos(phi) * 6 * nu ** 3)
282        XII = (5 + 28 * tan(phi) ** 2 + 24 * tan(phi) ** 4) / (cos(phi) * 120 * nu ** 5)
283        XIIA = (61 + 662 * tan(phi) ** 2 + 1320 * tan(phi) ** 4 + 720 * tan(phi) ** 6) / (cos(phi) * 5040 * nu ** 7)
284
285        phi = phi - VII * (E - E_0) ** 2 + VIII * (E - E_0) ** 4 - IX * (E - E_0) ** 6
286        lam = lam_0 + X * (E - E_0) - XI * (E - E_0) ** 3 + XII * (E - E_0) ** 5 - XIIA * (E - E_0) ** 7
287        phi, lam = OSGB36toWGS84(phi, lam, rads=True)
288        my_phi.append(rad2dms(phi)[0])
289        my_lam.append(rad2dms(lam)[0])
290
291    return np.array(my_phi), np.array(my_lam)

Get OSGB36 easting/northing from GPS latitude and longitude pairs.

Parameters

east: float/arraylike OSGB36 easting value(s) (in m). north: float/arrayling OSGB36 easting value(s) (in m). rads: bool (optional) If true, specifies ouput is is radians. dms: bool (optional) If true, output is in degrees/minutes/seconds. Incompatible with rads option.

Returns

numpy.ndarray GPS (i.e. WGS84 datum) latitude value(s). numpy.ndarray GPS (i.e. WGS84 datum) longitude value(s).

Examples

>>> get_gps_lat_long_from_easting_northing([429157], [623009])
(array([55.49999916]), array([-1.54000861]))
